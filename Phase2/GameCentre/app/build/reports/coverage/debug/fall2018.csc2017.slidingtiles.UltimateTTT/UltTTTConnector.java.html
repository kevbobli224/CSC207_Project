    /**
     * Gets the index of the image button
     *
     * @param b the image button
     * @return the index of the image button
     */
    int getIndex(ImageButton b) {
        for (int i = 0; i < ImageButtons.length; i++) {
            if (ImageButtons[i].equals(b)) {
                return i;
            }
        }
        if (b.equals(breset))
            return 100;
        if (b.equals(bundo))
            return 200;
        return -1;
